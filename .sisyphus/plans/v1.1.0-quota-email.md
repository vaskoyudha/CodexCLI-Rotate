- **Usage/Quota**: `GET https://chatgpt.com/backend-api/wham/usage`
  - Headers: `Authorization: Bearer <access_token>`, `chatgpt-account-id: <account_id>`
  - Returns: `{ plan_type, rate_limit: { primary_window: { used_percent, limit_window_seconds, reset_at }, secondary_window: {...} }, credits: { has_credits, unlimited, balance } }`

  - Returns: `{ id_token, access_token, refresh_token, ... }`

```bash
extract_jwt_claims() {
  # Decode the full JWT payload and return the raw JSON.
  local token="$1"
  [[ -n "${token}" ]] || { printf "{}"; return 0; }

  local payload_b64 payload_json
  payload_b64="${token#*.}"
  payload_b64="${payload_b64%%.*}"
  payload_json="$(decode_base64url "${payload_b64}")"
  [[ -n "${payload_json}" ]] || { printf "{}"; return 0; }

  # Validate it's parseable JSON
  if printf "%s" "${payload_json}" | jq -e '.' >/dev/null 2>&1; then
    printf "%s" "${payload_json}"
  else
    printf "{}"
  fi
}

extract_plan_type() {
  local token="$1"
  local claims
  claims="$(extract_jwt_claims "${token}")"
  printf "%s" "${claims}" | jq -r '.["https://api.openai.com/auth"].chatgpt_plan_type // empty' 2>/dev/null || true
}

extract_account_id_from_token() {
  local token="$1"
  local claims
  claims="$(extract_jwt_claims "${token}")"
  printf "%s" "${claims}" | jq -r '.["https://api.openai.com/auth"].chatgpt_account_id // empty' 2>/dev/null || true
}

get_account_tokens() {
  # Read tokens from an account's credential file.
  # Returns: JSON object with id_token, access_token, refresh_token, account_id
  local alias="$1"
  local cred_file="${CREDENTIALS_DIR}/${alias}.json"
  [[ -f "${cred_file}" ]] || { printf "{}"; return 0; }

  # Check if API key mode (no tokens)
  if jq -e 'has("OPENAI_API_KEY")' "${cred_file}" >/dev/null 2>&1; then
    printf '{"mode":"api_key"}'
    return 0
  fi

  jq -r '.tokens // {}' "${cred_file}" 2>/dev/null || printf "{}"
}

get_account_email() {
  local alias="$1"
  local tokens id_token
  tokens="$(get_account_tokens "${alias}")"
  id_token="$(printf "%s" "${tokens}" | jq -r '.id_token // empty' 2>/dev/null || true)"
  extract_email_from_id_token "${id_token}"
}

get_account_plan() {
  local alias="$1"
  local tokens id_token
  tokens="$(get_account_tokens "${alias}")"
  id_token="$(printf "%s" "${tokens}" | jq -r '.id_token // empty' 2>/dev/null || true)"
  extract_plan_type "${id_token}"
}
```

```bash
###############################################################################
# Usage/quota API helpers
###############################################################################

OPENAI_USAGE_URL="https://chatgpt.com/backend-api/wham/usage"
OPENAI_REFRESH_URL="https://auth.openai.com/oauth/token"
OPENAI_CLIENT_ID="app_EMoamEEZ73f0CkXaXp7hrann"

query_usage_api() {
  # Query the OpenAI usage API for a given account.
  # Returns: raw JSON response or empty string on failure.
  local alias="$1"
  local tokens access_token account_id

  tokens="$(get_account_tokens "${alias}")"

  # API key mode — no usage API available
  if [[ "$(printf "%s" "${tokens}" | jq -r '.mode // empty')" == "api_key" ]]; then
    printf ""
    return 0
  fi

  access_token="$(printf "%s" "${tokens}" | jq -r '.access_token // empty')"
  account_id="$(printf "%s" "${tokens}" | jq -r '.account_id // empty')"

  [[ -n "${access_token}" ]] || { printf ""; return 0; }

  local response http_code
  response="$(curl -sS --max-time 10 \
    -H "Authorization: Bearer ${access_token}" \
    -H "User-Agent: codex-cli/1.0.0" \
    ${account_id:+-H "chatgpt-account-id: ${account_id}"} \
    -w "\n%{http_code}" \
    "${OPENAI_USAGE_URL}" 2>/dev/null || true)"

  # Split response body and HTTP code
  http_code="$(printf "%s" "${response}" | tail -n1)"
  response="$(printf "%s" "${response}" | sed '$d')"

  if [[ "${http_code}" == "200" ]] && printf "%s" "${response}" | jq -e '.' >/dev/null 2>&1; then
    printf "%s" "${response}"
  else
    printf ""
  fi
}

refresh_account_token() {
  # Refresh tokens for a given account using refresh_token.
  # Updates the credential file in-place on success.
  # Returns: 0 on success, 1 on failure.
  local alias="$1"
  local cred_file="${CREDENTIALS_DIR}/${alias}.json"
  [[ -f "${cred_file}" ]] || return 1

  local tokens refresh_token
  tokens="$(get_account_tokens "${alias}")"

  if [[ "$(printf "%s" "${tokens}" | jq -r '.mode // empty')" == "api_key" ]]; then
    log_warn "Account '${alias}' uses API key mode — token refresh not applicable"
    return 1
  fi

  refresh_token="$(printf "%s" "${tokens}" | jq -r '.refresh_token // empty')"
  [[ -n "${refresh_token}" ]] || {
    log_warn "Account '${alias}' has no refresh_token"
    return 1
  }

  local response http_code
  response="$(curl -sS --max-time 15 \
    -X POST "${OPENAI_REFRESH_URL}" \
    -H "Content-Type: application/x-www-form-urlencoded" \
    -d "grant_type=refresh_token&refresh_token=${refresh_token}&client_id=${OPENAI_CLIENT_ID}" \
    -w "\n%{http_code}" \
    2>/dev/null || true)"

  http_code="$(printf "%s" "${response}" | tail -n1)"
  response="$(printf "%s" "${response}" | sed '$d')"

  if [[ "${http_code}" != "200" ]]; then
    log_warn "Token refresh failed for '${alias}' (HTTP ${http_code})"
    return 1
  fi

  # Validate response has tokens
  local new_id_token new_access_token new_refresh_token
  new_id_token="$(printf "%s" "${response}" | jq -r '.id_token // empty' 2>/dev/null || true)"
  new_access_token="$(printf "%s" "${response}" | jq -r '.access_token // empty' 2>/dev/null || true)"
  new_refresh_token="$(printf "%s" "${response}" | jq -r '.refresh_token // empty' 2>/dev/null || true)"

  if [[ -z "${new_access_token}" ]]; then
    log_warn "Token refresh for '${alias}' returned no access_token"
    return 1
  fi

  # Update credential file — preserve existing fields, update tokens
  local tmp_file
  tmp_file="$(mktemp "${cred_file}.tmp.XXXXXX")"

  jq --arg id "${new_id_token}" \
     --arg access "${new_access_token}" \
     --arg refresh "${new_refresh_token:-${refresh_token}}" \
     '.tokens.id_token = $id | .tokens.access_token = $access | .tokens.refresh_token = $refresh' \
     "${cred_file}" > "${tmp_file}" 2>/dev/null

  if [[ $? -eq 0 ]] && jq -e '.' "${tmp_file}" >/dev/null 2>&1; then
    chmod 600 "${tmp_file}"
    mv -f "${tmp_file}" "${cred_file}"
    return 0
  else
    rm -f "${tmp_file}"
    log_warn "Failed to update credential file for '${alias}'"
    return 1
  fi
}

parse_usage_percent() {
  # Extract primary window used_percent from usage JSON.
  local usage_json="$1"
  [[ -n "${usage_json}" ]] || { printf "0"; return 0; }
  local pct
  pct="$(printf "%s" "${usage_json}" | jq -r '.rate_limit.primary_window.used_percent // 0' 2>/dev/null || true)"
  [[ -n "${pct}" && "${pct}" != "null" ]] || pct="0"
  printf "%s" "${pct}"
}

parse_usage_reset_at() {
  # Extract primary window reset_at ISO timestamp from usage JSON.
  local usage_json="$1"
  [[ -n "${usage_json}" ]] || { printf ""; return 0; }
  printf "%s" "${usage_json}" | jq -r '.rate_limit.primary_window.reset_at // empty' 2>/dev/null || true
}

format_usage_bar() {
  # Render a text-based usage bar: [████████░░] 80%
  local percent="$1"
  local width="${2:-10}"
  # Ensure percent is an integer
  percent="${percent%%.*}"
  [[ "${percent}" =~ ^[0-9]+$ ]] || percent=0
  (( percent > 100 )) && percent=100

  local filled=$(( percent * width / 100 ))
  local empty=$(( width - filled ))

  local bar=""
  local i
  for (( i=0; i<filled; i++ )); do bar+="█"; done
  for (( i=0; i<empty; i++ )); do bar+="░"; done

  local color="${GREEN}"
  if (( percent >= 90 )); then
    color="${RED}"
  elif (( percent >= 70 )); then
    color="${YELLOW}"
  fi

  printf "%b[%s]%b %3d%%" "${color}" "${bar}" "${RESET}" "${percent}"
}
```

```bash
cmd_quota() {
  require_initialized
  load_config

  command -v curl >/dev/null 2>&1 || die "curl is required for quota checking"

  local aliases_to_check=()

  if (( $# > 0 )); then
    local alias="$1"
    validate_alias "${alias}" || die "Invalid alias '${alias}'."
    account_exists "${alias}" || die "Alias '${alias}' not found"
    aliases_to_check=("${alias}")
  else
    mapfile -t aliases_to_check < <(get_order)
  fi

  if (( ${#aliases_to_check[@]} == 0 )); then
    die "No accounts configured. Use '${SCRIPT_NAME} add' first."
  fi

  local alias email plan usage_json primary_pct primary_reset secondary_pct secondary_reset
  local credits_balance credits_unlimited plan_type_from_api

  for alias in "${aliases_to_check[@]}"; do
    email="$(get_account_email "${alias}")"
    plan="$(get_account_plan "${alias}")"

    printf "\n%b━━━ %s ━━━%b\n" "${BLUE}" "${alias}" "${RESET}"
    printf "  Email:  %s\n" "${email:-unknown}"
    printf "  Plan:   %s\n" "${plan:-unknown}"

    usage_json="$(query_usage_api "${alias}")"

    if [[ -z "${usage_json}" ]]; then
      printf "  %b⚠ Could not fetch usage data (token may be expired — try: %s refresh %s)%b\n" \
        "${YELLOW}" "${SCRIPT_NAME}" "${alias}" "${RESET}"
      continue
    fi

    plan_type_from_api="$(printf "%s" "${usage_json}" | jq -r '.plan_type // empty' 2>/dev/null || true)"
    [[ -n "${plan_type_from_api}" ]] && printf "  Plan (API): %s\n" "${plan_type_from_api}"

    # Primary window (5-hour)
    primary_pct="$(printf "%s" "${usage_json}" | jq -r '.rate_limit.primary_window.used_percent // 0' 2>/dev/null || true)"
    primary_reset="$(printf "%s" "${usage_json}" | jq -r '.rate_limit.primary_window.reset_at // empty' 2>/dev/null || true)"

    printf "  5-Hour Usage:  "
    format_usage_bar "${primary_pct}"
    if [[ -n "${primary_reset}" ]]; then
      local reset_epoch now_epoch_val remaining_secs
      reset_epoch="$(date -d "${primary_reset}" +%s 2>/dev/null || date -j -f "%Y-%m-%dT%H:%M:%S" "${primary_reset%%.*}" +%s 2>/dev/null || echo 0)"
      now_epoch_val="$(now_epoch)"
      if (( reset_epoch > now_epoch_val )); then
        remaining_secs=$(( reset_epoch - now_epoch_val ))
        printf " (resets in %s)" "$(format_duration "${remaining_secs}")"
      fi
    fi
    printf "\n"

    # Secondary window (weekly)
    secondary_pct="$(printf "%s" "${usage_json}" | jq -r '.rate_limit.secondary_window.used_percent // 0' 2>/dev/null || true)"
    secondary_reset="$(printf "%s" "${usage_json}" | jq -r '.rate_limit.secondary_window.reset_at // empty' 2>/dev/null || true)"

    printf "  Weekly Usage:  "
    format_usage_bar "${secondary_pct}"
    if [[ -n "${secondary_reset}" ]]; then
      local reset_epoch2 now_epoch_val2 remaining_secs2
      reset_epoch2="$(date -d "${secondary_reset}" +%s 2>/dev/null || date -j -f "%Y-%m-%dT%H:%M:%S" "${secondary_reset%%.*}" +%s 2>/dev/null || echo 0)"
      now_epoch_val2="$(now_epoch)"
      if (( reset_epoch2 > now_epoch_val2 )); then
        remaining_secs2=$(( reset_epoch2 - now_epoch_val2 ))
        printf " (resets in %s)" "$(format_duration "${remaining_secs2}")"
      fi
    fi
    printf "\n"

    # Credits
    credits_balance="$(printf "%s" "${usage_json}" | jq -r '.credits.balance // empty' 2>/dev/null || true)"
    credits_unlimited="$(printf "%s" "${usage_json}" | jq -r '.credits.unlimited // empty' 2>/dev/null || true)"
    if [[ "${credits_unlimited}" == "true" ]]; then
      printf "  Credits:  %b∞ Unlimited%b\n" "${GREEN}" "${RESET}"
    elif [[ -n "${credits_balance}" && "${credits_balance}" != "null" ]]; then
      printf "  Credits:  %s\n" "${credits_balance}"
    fi
  done
  printf "\n"
}
```

```bash
get_best_quota_account() {
  # Attempt to find the account with the lowest primary window usage.
  # Falls back to round-robin (get_next_available) if quota data is unavailable.
  # Only queries if curl is available and accounts use browser login.
  command -v curl >/dev/null 2>&1 || {
    get_next_available
    return
  }

  mapfile -t all_aliases < <(get_order)
  local total="${#all_aliases[@]}"
  (( total > 0 )) || { printf ""; return 0; }

  local best_alias="" best_pct=101
  local alias tokens usage_json pct

  for alias in "${all_aliases[@]}"; do
    [[ -f "${CREDENTIALS_DIR}/${alias}.json" ]] || continue
    is_on_cooldown "${alias}" && continue

    tokens="$(get_account_tokens "${alias}")"
    # Skip API key accounts (no usage API)
    if [[ "$(printf "%s" "${tokens}" | jq -r '.mode // empty')" == "api_key" ]]; then
      # API key accounts are usable but we can't check their quota
      if [[ -z "${best_alias}" ]]; then
        best_alias="${alias}"
      fi
      continue
    fi

    usage_json="$(query_usage_api "${alias}" 2>/dev/null)"
    if [[ -z "${usage_json}" ]]; then
      # Can't reach API — treat as unknown, eligible as fallback
      if [[ -z "${best_alias}" ]]; then
        best_alias="${alias}"
      fi
      continue
    fi

    pct="$(parse_usage_percent "${usage_json}")"
    pct="${pct%%.*}"
    [[ "${pct}" =~ ^[0-9]+$ ]] || pct=0

    # Skip accounts at 90%+ usage
    if (( pct >= 90 )); then
      continue
    fi

    if (( pct < best_pct )); then
      best_pct="${pct}"
      best_alias="${alias}"
    fi
  done

  if [[ -n "${best_alias}" ]]; then
    printf "%s" "${best_alias}"
  else
    # All accounts exhausted or at 90%+ — fall back to round-robin
    get_next_available
  fi
}
```
